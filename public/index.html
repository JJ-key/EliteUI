About

A mobile-optimized UI library for Roblox

EliteUI is a modern UI library designed for Roblox mobile scripts. It provides a sharp purple theme, touch-friendly controls, and a simple API for building script hubs and tools.

Mobile Optimized

Built from the ground up for mobile devices with touch-friendly controls and responsive layouts.

Purple & Black Theme

Modern dark theme with smooth animations and clean typography.

Gesture Support

Hold 3 fingers for 0.8 seconds to toggle UI visibility on mobile - designed to not trigger during gameplay.

Simple API

Intuitive methods for creating windows, tabs, buttons, toggles, sliders, and more.

Components

Interactive UI elements with live previews and usage code

Window

The main container for your UI. Creates a draggable window with tabs. Supports optional Key System for access control and Snow Effect for a festive animated background.

Lua 
local Window = EliteUI:CreateWindow({
    Name = "My Script Hub",
    LoadingTitle = "Loading...",
    Subtitle = "v1.0",
    
    -- Key System (optional)
    KeySystem = {
        Enabled = true,
        URL = "https://pastebin.com/raw/YOURKEY",
        Title = "Key Required",
        Subtitle = "Enter your key to continue",
        NoteName = "Get Key",
        NoteURL = "https://yoursite.com/getkey",
        SaveKey = true,
        FileName = "key.txt"
    },
    
    -- Snow Effect (optional)
    SnowEffect = {
        Enabled = true,
        ParticleCount = 50,  -- Number of snowflakes (default: 50)
        Speed = 1,           -- Fall speed multiplier (default: 1)
        Glow = true          -- Enable glow effect on snowflakes (default: true)
    }
})

Button

Clickable button that executes a callback function.

Lua 
MainTab:CreateButton({
    Name = "Click Me",
    Callback = function()

    end
})

Toggle

On/off switch that returns its state to a callback.

Enable Feature
Lua 
MainTab:CreateToggle({

    CurrentValue = false,

Invisible Button

A draggable button that can be used to toggle the UI. Completely invisible when UI is closed, visible when UI is open. Supports custom images.

Lua - Basic 
local InvisibleBtn = Window:CreateInvisibleButton({
    Size = 50,
    TapCount = 3,
    SavePosition = true
})
Lua - With Custom Image 
local InvisibleBtn = Window:CreateInvisibleButton({
    Size = 50,
    TapCount = 3,
    SavePosition = true,
    Image = "rbxassetid://123456789",
    Callback = function()
        Window:Toggle()
    end
})

-- Update image later
InvisibleBtn:SetImage("star")

-- Methods
InvisibleBtn:SetSize(60)
InvisibleBtn:SetPosition(0.5, 0.1)
InvisibleBtn:SetTapCount(2)
InvisibleBtn:Destroy()

Default Settings Tab

Automatically creates a complete settings tab with theme customization, UI scale, size presets, notifications, and more. This is optional but saves a lot of time!

Lua - Quick One-Liner 
-- Creates a complete settings tab with all options
local SettingsTab = Window:CreateDefaultSettingsTab()
Lua - With Custom Options 
local SettingsTab = Window:CreateDefaultSettingsTab({
    TabName = "Settings",           -- Tab name (default: "Settings")
    TabIcon = "settings",           -- Tab icon
    ShowThemeCustomization = true,  -- Color picker & theme presets
    ShowUIScale = true,             -- UI scale slider
    ShowUISizePresets = true,       -- Size preset dropdown
    ShowNotifications = true,       -- Notification toggle & test
    ShowDataManagement = true,      -- Clear settings button
    ShowControls = true,            -- Toggle key info
    ShowInfo = true,                -- Version & about
    Version = "1.0.0",              -- Your script version
    AboutText = "My custom script.",
    CustomThemePresets = {          -- Optional: custom themes
        ["Ocean"] = Color3.fromRGB(6, 182, 212),
        ["Forest"] = Color3.fromRGB(34, 197, 94)
    }
})

-- You can still add custom elements to the tab!
SettingsTab:CreateSection("My Custom Settings")
SettingsTab:CreateToggle({
    Name = "Debug Mode",
    CurrentValue = false,
    Callback = function(Value)
        print("Debug:", Value)
    end
})

Example

Complete example script using all components

Complete Example Script 
-- Load EliteUI
local EliteUI = loadstring(game:HttpGet("https://raw.githubusercontent.com/JJ-key/EliteUI/main/Loader"))()

-- Create Window
local Window = EliteUI:CreateWindow({
    Name = "EliteUI Demo Hub",
    LoadingTitle = "Loading...",
})

-- Create Tabs
local MainTab = Window:CreateTab("Main", "home")
local CombatTab = Window:CreateTab("Combat", "sword")
local SettingsTab = Window:CreateTab("Settings", "settings")

-- Main Tab
MainTab:CreateSection("Welcome")

MainTab:CreateParagraph({
    Title = "EliteUI Library",
    Content = "A modern, mobile-optimized UI library for Roblox."
})

MainTab:CreateButton({
    Name = "Show Notification",
    Callback = function()
        Window:Notify({
            Title = "Success!",
            Content = "Button clicked successfully.",
            Duration = 3
        })
    end
})

local StatusLabel = MainTab:CreateLabel("Status: Ready")

MainTab:CreateTextBox({
    Name = "Username",
    Placeholder = "Enter username...",
    CurrentValue = "",
    Callback = function(Text, EnterPressed)
        StatusLabel:SetText("Username: " .. Text)
    end
})

-- Combat Tab
CombatTab:CreateSection("Combat Features")

CombatTab:CreateToggle({
    Name = "Kill Aura",
    CurrentValue = false,
    Callback = function(Value)
        print("Kill Aura:", Value)
    end
})

CombatTab:CreateSlider({
    Name = "Attack Range",
    Min = 5,
    Max = 50,
    CurrentValue = 15,
    Callback = function(Value)
        print("Range:", Value)
    end
})

CombatTab:CreateDropdown({
    Name = "Target Mode",
    Options = {"Nearest", "Lowest HP", "Random", "Manual"},
    CurrentValue = "Nearest",
    Callback = function(Option)
        print("Selected:", Option)
    end
})

CombatTab:CreateKeybind({
    Name = "Toggle Aura",
    CurrentKeybind = Enum.KeyCode.E,
    Callback = function(NewKey)
        print("New keybind:", NewKey.Name)
    end,
    HoldCallback = function()
        print("Keybind pressed!")
    end
})

-- Settings Tab
SettingsTab:CreateSection("Customization")

SettingsTab:CreateColorPicker({
    Name = "Accent Color",
    CurrentColor = Color3.fromRGB(139, 92, 246),
    Callback = function(Color)
        print("Color:", Color)
    end
})

SettingsTab:CreateDropdown({
    Name = "Theme",
    Options = {"Dark Purple", "Violet", "Dark Green"},
    CurrentValue = "Dark Purple",
    Callback = function(Theme)
        print("Theme:", Theme)
    end
})

SettingsTab:CreateToggle({
    Name = "Show Notifications",
    CurrentValue = true,
    Callback = function(Value)
        print("Notifications:", Value)
    end
})

SettingsTab:CreateLabel("EliteUI v2.0.0")

SettingsTab:CreateButton({
    Name = "Test Notification",
    Callback = function()
        Window:Notify({
            Title = "Hello!",
            Content = "This is a notification from EliteUI.",
            Duration = 4
        })
    end
})
