About EliteUI

A modern, mobile-optimized UI library for Roblox

EliteUI is a sleek, modern UI library designed specifically for Roblox mobile scripts. It features a beautiful dark blue theme, intuitive mobile gestures, and easy-to-use components.

Available Components

Window

The main container for your UI

local Window = EliteUI:CreateWindow({
    Name = "My Script Hub",
    LoadingTitle = "Loading...",
})

Tab

Create navigation tabs to organize your UI

local MainTab = Window:CreateTab("Main", "home")
local SettingsTab = Window:CreateTab("Settings", "settings")

Section

Group related elements together

MainTab:CreateSection("Welcome")

Button

Clickable button with callback

MainTab:CreateButton({
    Name = "Click Me",
    Callback = function()
        print("Button clicked!")
    end
})

Toggle

On/Off switch with state callback

MainTab:CreateToggle({
    Name = "Enable Feature",
    CurrentValue = false,
    Callback = function(Value)
        print("Toggle:", Value)
    end
})

Slider

Value slider with min/max range

MainTab:CreateSlider({
    Name = "Speed",
    Min = 0,
    Max = 100,
    CurrentValue = 50,
    Callback = function(Value)
        print("Speed:", Value)
    end
})

Dropdown

Selection menu with multiple options

MainTab:CreateDropdown({
    Name = "Select Mode",
    Options = {"Option 1", "Option 2", "Option 3"},
    CurrentValue = "Option 1",
    Callback = function(Option)
        print("Selected:", Option)
    end
})

TextBox

Text input field with placeholder

MainTab:CreateTextBox({
    Name = "Username",
    Placeholder = "Enter text...",
    CurrentValue = "",
    Callback = function(Text, EnterPressed)
        print("Input:", Text)
    end
})

Label

Dynamic text label that can be updated

local MyLabel = MainTab:CreateLabel("Hello World!")
MyLabel:SetText("Updated text!")

Paragraph

Title with description text block

MainTab:CreateParagraph({
    Title = "Information",
    Content = "This is a longer description text."
})

Keybind

Customizable keyboard shortcut picker

MainTab:CreateKeybind({
    Name = "Toggle Feature",
    CurrentKeybind = Enum.KeyCode.E,
    Callback = function(NewKey)
        print("New keybind:", NewKey.Name)
    end,
    HoldCallback = function()
        print("Key pressed!")
    end
})

Color Picker

Choose from preset colors (click to cycle)

local MyColor = MainTab:CreateColorPicker({
    Name = "Accent Color",
    CurrentColor = Color3.fromRGB(59, 130, 246),
    Callback = function(Color)
        print("Color:", Color)
    end
})
MyColor:SetColor(Color3.fromRGB(255, 0, 0))

Notification

Show toast notifications to the user

Window:Notify({
    Title = "Success!",
    Content = "Operation completed successfully.",
    Duration = 3
})

Complete Example

Example Script 
local EliteUI = loadstring(game:HttpGet("YOUR_URL/Loader"))()

local Window = EliteUI:CreateWindow({
    Name = "EliteUI Demo Hub",
    LoadingTitle = "Loading...",
})

local MainTab = Window:CreateTab("Main", "home")
local CombatTab = Window:CreateTab("Combat", "sword")
local SettingsTab = Window:CreateTab("Settings", "settings")

-- Main Tab
MainTab:CreateSection("Welcome")

MainTab:CreateParagraph({
    Title = "EliteUI Library",
    Content = "A modern, mobile-optimized UI library for Roblox."
})

MainTab:CreateButton({
    Name = "Show Notification",
    Callback = function()
        Window:Notify({
            Title = "Success!",
            Content = "Button clicked successfully.",
            Duration = 3
        })
    end
})

MainTab:CreateLabel("This is a dynamic label")

MainTab:CreateTextBox({
    Name = "Username",
    Placeholder = "Enter username...",
    Callback = function(Text)
        print("Input:", Text)
    end
})

-- Combat Tab
CombatTab:CreateSection("Combat Features")

CombatTab:CreateToggle({
    Name = "Kill Aura",
    CurrentValue = false,
    Callback = function(Value)
        print("Kill Aura:", Value)
    end
})

CombatTab:CreateSlider({
    Name = "Attack Range",
    Min = 5,
    Max = 50,
    CurrentValue = 15,
    Callback = function(Value)
        print("Range:", Value)
    end
})

CombatTab:CreateDropdown({
    Name = "Target Mode",
    Options = {"Nearest", "Lowest HP", "Random"},
    CurrentValue = "Nearest",
    Callback = function(Option)
        print("Selected:", Option)
    end
})

CombatTab:CreateKeybind({
    Name = "Toggle Aura",
    CurrentKeybind = Enum.KeyCode.E,
    HoldCallback = function()
        print("Keybind pressed!")
    end
})

-- Settings Tab
SettingsTab:CreateSection("Customization")

SettingsTab:CreateColorPicker({
    Name = "Accent Color",
    CurrentColor = Color3.fromRGB(59, 130, 246),
    Callback = function(Color)
        print("Color:", Color)
    end
})
